package
{
    import away3d.containers.ObjectContainer3D;
    import away3d.containers.View3D;
    import away3d.core.base.Mesh;
    import away3d.core.base.Object3D;
    import away3d.core.base.Segment;
    import away3d.core.base.Vertex;
    import away3d.core.draw.ScreenVertex;
    import away3d.core.math.Number3D;
    import away3d.events.MouseEvent3D;
    import away3d.events.ViewEvent;
    import away3d.materials.ColorMaterial;
    import away3d.materials.WireframeMaterial;
    import away3d.primitives.Plane;
    
    import flash.display.MovieClip;
    import flash.display.Sprite;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.events.ProgressEvent;

    public class InteractiveSegment extends ObjectContainer3D
    {
        private var _view:View3D;
        private var _segment:Segment;
        private var _segmentHolder:Mesh;
        private var _segmentPoints:Array;
        private var _markers3d:Array;
        private var _markers2d:Array;
        private var _capturePlane:Plane;
        private var _tracker3d:Object3D;
        private var _draggingMarkerIndex:int = -1;
        
        public function get path():Segment
        {
            return _segment;    
        }
        
        public function InteractiveSegment(view:View3D)
        {
            super();
            
            _view = view;
            
            _segmentHolder = new Mesh();
            addChild(_segmentHolder);
            
            createSegmentPoints();
            createSegment();
            createMarkers2d();
            createMarkers3d();
            createCapturePlaneAndTracker();
            
            _view.addEventListener(MouseEvent.MOUSE_UP, viewMouseUpHandler);
            _view.addEventListener(ViewEvent.UPDATE_SCENE, alignMarkers);
        }
        
        private function createCapturePlaneAndTracker():void
        {
            _capturePlane = new Plane();
            _capturePlane.rotationX = 90;
            _capturePlane.width = _capturePlane.height = 2000;
            _capturePlane.material = new ColorMaterial(0xFFFFFF, {alpha:0.01});
            _capturePlane.addOnMouseMove(capturePlaneMouse3dMoveHandler);
            addChild(_capturePlane);
            
            _tracker3d = new Object3D();
            addChild(_tracker3d);
        }
        
        private function capturePlaneMouse3dMoveHandler(evt:MouseEvent3D):void
        {
            _tracker3d.position = new Number3D(evt.sceneX, evt.sceneY, evt.sceneZ);
        }
        
        private function createSegmentPoints():void
        {
            var dW:Number = 100;
            var dH:Number = 250;
            var dD:Number = 1000;
            var q:Number = 10;
            var centerOffsetX:Number = dW*q/2;
            _segmentPoints = [new Number3D(-centerOffsetX, 0, 0)];
            for(var i:uint = 0; i<q; i++)
            {
                var H:Number = dH*Math.random() - dH*Math.random();
                var D:Number = dH*Math.random() - dH*Math.random();
                _segmentPoints.push(new Number3D((i+1)*dW - centerOffsetX, H, D));
            }
        }
        
        private function createSegment():void
        {
            if(_segment)
                _segmentHolder.removeSegment(_segment);
            
            _segment = new Segment();
            _segment.continuousCurve(_segmentPoints);
            _segment.material = new WireframeMaterial(0xFFFFFF, {alpha:0.5});
            _segmentHolder.addSegment(_segment);
        }
        
        private function createMarkers2d():void
        {
            _markers2d = [];
            
            var index:uint;
            for each(var vertex:Vertex in _segmentHolder.vertices)
            {
                var marker2d:MovieClip = new MovieClip();
                marker2d.vertexIndex = index;
                marker2d.graphics.beginFill(0xFF0000, 0.33);
                marker2d.graphics.drawCircle(0, 0, 8);
                marker2d.graphics.endFill();
                marker2d.useHandCursor = marker2d.buttonMode = true;
                marker2d.visible = index%2 != 0;
                marker2d.addEventListener(MouseEvent.MOUSE_DOWN, markerMouseDownHandler);
                _view.addChild(marker2d);
                _markers2d.push(marker2d);
                
                index++;
            }
        }
        
        private function createMarkers3d():void
        {
            _markers3d = [];
            
            var index:uint;
            for each(var vertex:Vertex in _segmentHolder.vertices)
            {
                var marker3d:Object3D = new Object3D();
                marker3d.position = new Number3D(vertex.x, vertex.y, vertex.z);
                _markers3d.push(marker3d);
                
                index++;
            }
        }
        
        private function markerMouseDownHandler(evt:MouseEvent):void
        {
            var marker2d:MovieClip = evt.target as MovieClip;
            var marker3d:Object3D = _markers3d[marker2d.vertexIndex];
            _capturePlane.position = marker3d.position;
            
            _draggingMarkerIndex = marker2d.vertexIndex;
        }
        
        private function viewMouseUpHandler(evt:MouseEvent):void
        {
            _draggingMarkerIndex = -1;
            
            this.dispatchEvent(new Event(Event.COMPLETE));
        }
        
        private function alignMarkers(evt:Event):void
        {
            if(!_markers2d || !_markers3d)
                return;
            
            for(var i:uint; i<_markers2d.length; i++)
            {
                var marker2d:Sprite = _markers2d[i];
                var marker3d:Object3D = _markers3d[i];
                
                var sv:ScreenVertex;
                if(_draggingMarkerIndex == i)
                {
                    var weirdIndex:uint = 0.5*i + 1;
                    if(_segmentPoints[weirdIndex])
                    {
                        _segmentPoints[weirdIndex].x = _tracker3d.x;
                        _segmentPoints[weirdIndex].y = _tracker3d.y;
                    }
                    else
                        continue;
                    
                    sv = _view.camera.screen(_tracker3d);
                    
                    createSegment();
                    createMarkers3d();
                    
                    this.dispatchEvent(new ProgressEvent(ProgressEvent.PROGRESS));
                }
                else
                    sv = _view.camera.screen(marker3d);
                
                marker2d.x = sv.x;
                marker2d.y = sv.y;
            }
        }
    }
}
