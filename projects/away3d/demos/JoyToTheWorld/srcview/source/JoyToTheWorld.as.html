package
{
    import away3d.cameras.*;
    import away3d.core.material.*;
    import away3d.core.math.Number3D;
    import away3d.core.render.BasicRenderer;
    import away3d.core.scene.*;
    import away3d.core.utils.Cast;
    import away3d.objects.Sphere;
    
    import com.adobe.viewsource.ViewSource;
    
    import flash.display.*;
    import flash.events.Event;
    import flash.events.MouseEvent;
    import flash.utils.getTimer;
    
    [SWF(backgroundColor="#000000", frameRate="60", quality="LOW")]
    
    public class JoyToTheWorld extends Sprite
    {
        [Embed(source="images/earth.jpg")]
        public static var EarthImage:Class;
        
        [Embed(source="images/Smiley-face.gif")]
        public static var SmileyImage:Class;
        
        [Embed(source="images/signature.swf")]
        public static var SignatureSwf:Class;
        
        protected var camera:HoverCamera3D;
        protected var view:View3D;
        protected var earth:Sphere;
        protected var scene:Scene3D;
        protected var Signature:MovieClip;
        
        private var active:Boolean = true;
        private var animation:Boolean = true;
        private var scroll:Boolean = true;
        private var time:int;  
        
        public var projectedMaterial:BitmapMaterialContainer;
        public var projectedMaterialArray:Array;
        public var projectedMaterial1:TransformBitmapMaterial;
        public var projectedMaterial2:TransformBitmapMaterial;
        public var projectedMaterial3:TransformBitmapMaterial;
        public var projectedMaterial4:TransformBitmapMaterial;
        public var projectedMaterial5:TransformBitmapMaterial;
        public var projectedMaterial6:TransformBitmapMaterial;
        
        public var projectionVector1:Number3D = new Number3D();
        public var projectionVector2:Number3D = new Number3D();
        public var projectionVector3:Number3D = new Number3D();
        public var projectionVector4:Number3D = new Number3D();
        public var projectionVector5:Number3D = new Number3D();
        public var projectionVector6:Number3D = new Number3D();
    
        public function JoyToTheWorld()
        {
            super();
            
            //take care of stage
            //stage.quality = StageQuality.LOW;
            stage.align = StageAlign.TOP_LEFT;
            stage.scaleMode = StageScaleMode.NO_SCALE;
            //stage.showDefaultContextMenu = false;
            stage.stageFocusRect = false;
            ViewSource.addMenuItem(this, "srcview/index.html")
            
            //setup hovercamera
            camera = new HoverCamera3D({zoom:3, focus:200, distance:2000});
            camera.tiltangle = 0;
            camera.targettiltangle = 0;
            camera.mintiltangle = -45;
            camera.maxtiltangle = 90;
            camera.yfactor = 1;
            
            //setup materials
            projectedMaterial1 = new TransformBitmapMaterial(Cast.bitmap(SmileyImage), {throughProjection:true});
            projectedMaterial2 = new TransformBitmapMaterial(Cast.bitmap(SmileyImage), {throughProjection:true});
            projectedMaterial3 = new TransformBitmapMaterial(Cast.bitmap(SmileyImage), {throughProjection:true});
            projectedMaterial4 = new TransformBitmapMaterial(Cast.bitmap(SmileyImage), {throughProjection:true});
            projectedMaterial5 = new TransformBitmapMaterial(Cast.bitmap(SmileyImage), {throughProjection:true});
            projectedMaterial6 = new TransformBitmapMaterial(Cast.bitmap(SmileyImage), {throughProjection:true});
            
            projectedMaterial = new BitmapMaterialContainer(512, 256, {materials:[
                            Cast.material(EarthImage),
                            projectedMaterial1,
                            projectedMaterial2,
                            projectedMaterial3,
                            projectedMaterial4,
                            projectedMaterial5,
                            projectedMaterial6
                            ]});
            //setup scene
            earth = new Sphere({material:projectedMaterial, radius:500, segmentsW:20, segmentsH:10});
            scene = new Scene3D(earth);
            
            //create view and add to displaylist
            view = new View3D({camera:camera, scene:scene, renderer:new BasicRenderer()});
            addChild(view);
            
            //add signature
            Signature = MovieClip(new SignatureSwf());
            addChild(Signature);
            
            stage.addEventListener(Event.ENTER_FRAME, onEnterFrame);
            stage.addEventListener(Event.RESIZE, onResize);
            stage.addEventListener(Event.ACTIVATE, onActivate);
            stage.addEventListener(Event.DEACTIVATE, onDeactivate);
            onResize(null);
        }
        
        public function onEnterFrame(event:Event):void
        {
            
            if (animation)
                time = getTimer();
               
            projectionVector1.x = (Math.sin(time/500));
            projectionVector1.y = 1;
            projectionVector1.z = (Math.cos(time/500));
            
            projectionVector2.x = (Math.cos(-time/500));
            projectionVector2.y = (Math.sin(time/500));
            projectionVector2.z = 1;
            
            projectionVector3.x = 1;
            projectionVector3.y = (Math.sin(-time/500));
            projectionVector3.z = (Math.cos(-time/500));
            
            projectionVector4.x = (Math.cos(-time/500));
            projectionVector4.y = 1;
            projectionVector4.z = (Math.sin(-time/500));
            
            projectionVector5.x = 1;
            projectionVector5.y = (Math.cos(-time/500));
            projectionVector5.z = (Math.sin(-time/500));
            
            projectionVector6.x = 1;
            projectionVector6.y = (Math.sin(time/500));
            projectionVector6.z = (Math.cos(-time/500));
            
            projectedMaterial1.rotation = time/1000;
            projectedMaterial2.rotation = time/500;
            projectedMaterial3.rotation = time/250;
            projectedMaterial4.rotation = time/125;
            projectedMaterial5.rotation = time/75;
            projectedMaterial6.rotation = time/250;
            
            projectedMaterial1.projectionVector = projectionVector1;
            projectedMaterial2.projectionVector = projectionVector2;
            projectedMaterial3.projectionVector = projectionVector3;
            projectedMaterial4.projectionVector = projectionVector4;
            projectedMaterial5.projectionVector = projectionVector5;
            projectedMaterial6.projectionVector = projectionVector6;
            
            if (!active)
                return;
               
            if (scroll)
            {
                if (view.mouseX > stage.stageWidth*0.40)
                    camera.targetpanangle -= 3;
                if (view.mouseX < -stage.stageWidth*0.40)
                    camera.targetpanangle += 3;

                if (view.mouseY > stage.stageHeight*0.40)
                    camera.targettiltangle -= 2;
                if (view.mouseY < -stage.stageHeight*0.40)
                    camera.targettiltangle += 2;
            }
            
            camera.hover();
            
            scene.updateTime(time);
            view.render();
        }

        
        private function onResize(event:Event):void 
        {
            view.x = stage.stageWidth / 2;
            view.y = stage.stageHeight / 2;
            Signature.y = stage.stageHeight - Signature.height;
        }
        
        private function onActivate(event:Event):void 
        {
            onResize(null);
            active = true;
        }

        private function onDeactivate(event:Event):void 
        {
            active = false;
        }
        

        private function onMouseWheel(event:MouseEvent):void
        {
            adjustzoom(event.delta)
        }

        private function adjustzoom(delta:int):void
        {
            camera.zoom = Math.min(25, Math.max(2, camera.zoom + delta / 3));
        }
    }
}
